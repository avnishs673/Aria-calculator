function calculateLand() { const shape = document.getElementById('shape').value; let area = 0; if (shape === 'irregular') { const e = parseFloat(document.getElementById('east').value) || 0; const w = parseFloat(document.getElementById('west').value) || 0; const n = parseFloat(document.getElementById('north').value) || 0; const s = parseFloat(document.getElementById('south').value) || 0; if (e && w && n && s) { area = (3 * (e + w) * (n + s)) / 4; } } else { const l = parseFloat(document.getElementById('lamb').value) || 0; const a = parseFloat(document.getElementById('aadhar').value) || 0; const v = parseFloat(document.getElementById('vikaran').value) || 0; if (l && a && v) { const sides = [l, a, v].sort((x, y) => x - y); area = ((sides[0] * sides[1]) / 2) * 3; } } if (area > 0) { // Logic for the Breakdown (Example: 532) let remaining = area; // 1. Calculate Full Bighas const beegha = Math.floor(remaining / 400); remaining = remaining % 400; // Leftover is 132 // 2. Calculate Full Biswa from leftover const biswa = Math.floor(remaining / 20); remaining = remaining % 20; // Leftover is 12 // 3. Final leftover is Biswansi const biswansi = remaining; document.getElementById('total').innerText = area.toFixed(2); document.getElementById('beegha').innerText = beegha; document.getElementById('biswa').innerText = biswa; document.getElementById('biswansi').innerText = biswansi.toFixed(2); document.getElementById('displayArea').style.display = 'block'; } else { alert("Please enter valid numbers!"); } }
